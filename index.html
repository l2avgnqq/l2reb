if(error){ console.log("SAVE ERROR:", error); } } /* ---------- TIME ---------- */ function updateTime(p){ if(!p.running) return; const last = new Date(p.last_update || Date.now()); const diff = Math.floor((Date.now() - last) / 1000); if(diff <= 0) return; const worked = Math.min(diff, p.time || 0); p.time -= worked; p.last_update = new Date().toISOString(); if(p.time <= 0){ p.time = 0; p.running = false; } } /* ---------- FORMAT ---------- */ function format(s){ const m = Math.floor(s / 60); const sec = s % 60; return m + ":" + sec.toString().padStart(2,"0"); } /* ---------- RENDER ---------- */ function render(){ const container = document.getElementById("players"); container.innerHTML = ""; container.appendChild(div); }); } /* ---------- ACTIONS ---------- */ async function add(){ const name = document.getElementById("name").value.trim(); const min = parseInt(document.getElementById("minutes").value); if(!name || !min) return; let p = data[name] || { name, time: 0, running: false, last_update: new Date().toISOString() }; updateTime(p); p.time += min * 60; p.running = true; p.last_update = new Date().toISOString(); await save(p); await load(); render(); } async function addTime(name){ const val = parseInt(document.getElementById("add_"+name).value); if(!val) return; let p = data[name]; updateTime(p); p.time += val * 60; p.running = true; p.last_update = new Date().toISOString(); await save(p); await load(); render(); } async function pause(name){ let p = data[name]; updateTime(p); p.running = false; await save(p); render(); } async function resume(name){ let p = data[name]; p.running = true; p.last_update = new Date().toISOString(); await save(p); render(); } async function delPlayer(name){ await db.from("timers").delete().eq("name", name); delete data[name]; render(); } /* ---------- INIT ---------- */ document.getElementById("addBtn").onclick = add; document.getElementById("addAllBtn").onclick = async () => { const min = parseInt(document.getElementById("allMinutes").value); p.running = true; p.last_update = new Date().toISOString(); await save(p); } render(); }; document.getElementById("clearAll").onclick = async () => { await db.from("timers").delete().neq("name",""); data = {}; render(); }; async function init(){ await load(); render(); } init(); setInterval(render, 1000);
